# install composer

curl -sS https://getcomposer.org/installer | php
#or
php -r "readfile('https://getcomposer.org/installer');" | php

# create project

php composer.phar create-project laravel/laravel --prefer-dist

# update libraries

php composer.phar update

# create tables

php artisan migrate:make create_users_table

php artisan migrate:make create_projects_table

php artisan migrate:make create_project_images_table

php artisan migrate:make add_token_to_user_table --table=users

php artisan migrate:make create_megabanners_table

php artisan migrate:make add_fields_to_projects_table --table=projects

php artisan migrate:make add_fields_to_megabanners_table --table=megabanners

# run migrations

php artisan migrate

#run seeds

php artisan db:seed

#create controller

php artisan controller:make ProjectController


# dump auto load

php composer.phar dump-autoload

# publish feed-reader config

php artisan config:publish awjudd/feed-reader
